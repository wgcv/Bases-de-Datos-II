Seleccionar la descripcion en otra tabla (JOIN)
SELECT B.department_id , B.DEPARTMENT_NAME, COUNT(*)
FROM EMPLOYEES A, DEPARTMENTS B
Where A.DEPARTMENT_ID = B.DEPARTMENT_ID
GROUP BY (B.DEPARTMENT_ID,B.DEPARTMENT_NAME)
ORDER BY 3 DESC


SELECT B.department_id , B.DEPARTMENT_NAME, COUNT(*), SUM(SALARY) SALARIO
FROM EMPLOYEES A, DEPARTMENTS B
Where A.DEPARTMENT_ID = B.DEPARTMENT_ID
GROUP BY (B.DEPARTMENT_ID,B.DEPARTMENT_NAME)
ORDER BY 3 DESC

Se agrega maximo minimo y promedio
SELECT 
B.department_id ,
B.DEPARTMENT_NAME, 
COUNT(*) AS , 
SUM(SALARY) SALARIO, 
MIN(SALARY) AS Minimo, 
MAX(SALARY) Maximo, 
AVG(SALARY) Promedio
FROM EMPLOYEES A, DEPARTMENTS B
Where A.DEPARTMENT_ID = B.DEPARTMENT_ID
GROUP BY (B.DEPARTMENT_ID,B.DEPARTMENT_NAME)
ORDER BY 3 DESC

Condicion:
	Solo Aquellos departamentos que tengan mas de 5 empleados
SELECT 
B.department_id ,
B.DEPARTMENT_NAME, 
COUNT(*) AS TOTAL, 
SUM(SALARY) SALARIO, 
MIN(SALARY) AS Minimo, 
MAX(SALARY) Maximo, 
AVG(SALARY) Promedio
FROM EMPLOYEES A, DEPARTMENTS B
Where A.DEPARTMENT_ID = B.DEPARTMENT_ID HAVING COUNT(*) > 5
GROUP BY (B.DEPARTMENT_ID,B.DEPARTMENT_NAME)
ORDER BY 3 DESC

Ademas mas de 150000
SELECT 
B.department_id ,
B.DEPARTMENT_NAME, 
COUNT(*) AS TOTAL, 
SUM(SALARY) SALARIO, 
MIN(SALARY) AS Minimo, 
MAX(SALARY) Maximo, 
AVG(SALARY) Promedio
FROM EMPLOYEES A, DEPARTMENTS B
Where A.DEPARTMENT_ID = B.DEPARTMENT_ID HAVING (COUNT(*) > 5 AND SUM(SALARY)>150000)
GROUP BY (B.DEPARTMENT_ID,B.DEPARTMENT_NAME)
ORDER BY 3 DESC
